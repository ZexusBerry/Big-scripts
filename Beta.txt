local function CreateDropdown(options, position)
	local dropdownFrame = Instance.new("Frame")
	dropdownFrame.Name = "CustomDropdownFrame"

	dropdownButton.Position = UDim2.new(0, 0, 0, 0)

	dropdownButton.Parent = dropdownFrame

		local optionButton = Instance.new("TextButton")
		optionButton.Name = "OptionButton"
		optionButton.Text = optionText
		optionButton.Size = UDim2.new(1, 0, 0, 30)

		optionButton.TextSize = 20

		optionButton.MouseButton1Click:Connect(function()
			dropdownButton.Text = optionText
			dropdownOptionsFrame.Visible = false
		end)
	end

	return dropdownButton

local function CreateSlider(minValue, maxValue, defaultValue, position)
	local sliderFrame = Instance.new("Frame")
	sliderFrame.Name = "CustomSliderFrame"
	sliderFrame.Size = UDim2.new(0.8, 0, 0.08, 0)
	sliderFrame.Position = position
	sliderFrame.AnchorPoint = Vector2.new(0.5, 0)
	sliderFrame.BackgroundColor3 = Color3.fromRGB(80, 80, 80)
	sliderFrame.BorderSizePixel = 2
	sliderFrame.Parent = Menu

	local sliderBar = Instance.new("Frame")
	sliderBar.Name = "SliderBar"
	sliderBar.Size = UDim2.new(1, 0, 0, 10)
	sliderBar.Position = UDim2.new(0, 0, 0.5, -5)
	sliderBar.BackgroundColor3 = Color3.fromRGB(100, 100, 100)
	sliderBar.BorderSizePixel = 0
	sliderBar.Parent = sliderFrame

	local slider = Instance.new("Frame")
	slider.Name = "CustomSlider"
	slider.Size = UDim2.new(0, 20, 0, 20)
	slider.BackgroundColor3 = Color3.fromRGB(200, 200, 200)
	slider.BorderSizePixel = 0
	slider.Parent = sliderBar

	local valueLabel = Instance.new("TextLabel")
	valueLabel.Name = "ValueLabel"
	valueLabel.Text = defaultValue
	valueLabel.Size = UDim2.new(0, 30, 0, 20)
	valueLabel.Position = UDim2.new(1, 5, 0.5, -10)
	valueLabel.BackgroundTransparency = 1
	valueLabel.Font = Enum.Font.SourceSans
	valueLabel.TextColor3 = Color3.new(1, 1, 1)
	valueLabel.TextSize = 16
	valueLabel.Parent = sliderFrame

	local dragging = false
	local function updateValue(mouseX)
		local sliderPosition = math.clamp(mouseX - sliderBar.AbsolutePosition.X, 0, sliderBar.AbsoluteSize.X - slider.AbsoluteSize.X)
		local percentage = sliderPosition / (sliderBar.AbsoluteSize.X - slider.AbsoluteSize.X)
		local value = math.floor(minValue + (maxValue - minValue) * percentage)
		slider.Position = UDim2.new(percentage, 0, 0.5, -10)
		valueLabel.Text = value
	end

	slider.MouseButton1Down:Connect(function()
		dragging = true
	end)

	game:GetService("UserInputService").InputEnded:Connect(function(input)
		if input.UserInputType == Enum.UserInputType.MouseButton1 then
			dragging = false
		end
	end)

	game:GetService("UserInputService").InputChanged:Connect(function(input)
		if dragging and input.UserInputType == Enum.UserInputType.MouseMovement then
			updateValue(input.Position.X)
		end
	end)

	updateValue(slider.AbsolutePosition.X)

	return sliderFrame
end

local function CreateToggleSwitch(defaultValue, position)
	local toggleFrame = Instance.new("Frame")
	toggleFrame.Name = "CustomToggleFrame"
	toggleFrame.Size = UDim2.new(0.2, 0, 0.08, 0)
	toggleFrame.Position = position
	toggleFrame.AnchorPoint = Vector2.new(0.5, 0)
	toggleFrame.BackgroundColor3 = Color3.fromRGB(80, 80, 80)
	toggleFrame.BorderSizePixel = 0
	toggleFrame.Parent = Menu

	local toggleKnob = Instance.new("TextButton")
	toggleKnob.Name = "ToggleKnob"
	toggleKnob.Text = ""
	toggleKnob.Size = UDim2.new(0.5, 0, 1, 0)
	toggleKnob.Position = UDim2.new(defaultValue and 0.5 or 0, 0, 0, 0)
	toggleKnob.BackgroundColor3 = Color3.fromRGB(200, 200, 200)
	toggleKnob.BorderColor3 = Color3.fromRGB(100, 100, 100)
	toggleKnob.BorderSizePixel = 2
	toggleKnob.Parent = toggleFrame

	local isToggled = defaultValue
	local function updateToggleVisual()
		toggleKnob:TweenPosition(UDim2.new(isToggled and 0.5 or 0, 0, 0, 0), "Out", "Quad", 0.3, true)
		toggleKnob.BackgroundColor3 = isToggled and Color3.fromRGB(255, 255, 255) or Color3.fromRGB(200, 200, 200)
	end

	toggleKnob.MouseButton1Click:Connect(function()
		isToggled = not isToggled
		updateToggleVisual()
	end)

	updateToggleVisual()

	return toggleFrame
end



--Creator Menu Function--


local button1 = CreateButton("Get Chest (Beta)", UDim2.new(0.5, 0, 0.07, 0))
local button2 = CreateButton("Teleport random Boat", UDim2.new(0.5, 0, 0.19, 0))
local button3 = CreateButton("Get up All Fruits", UDim2.new(0.5, 0, 0.31, 0))


--Creator Menu Function--




local workspacePath = game.Workspace

-- Список имен объектов, к которым нужно переместиться
local chestNames = {"Chest1", "Chest2", "Chest3"}

-- Функция для перемещения игрока к объектам по их именам
local function MoveToChests()
	for _, chestName in ipairs(chestNames) do
		local chest = workspacePath:FindFirstChild(chestName)
		if chest then
			game.Players.LocalPlayer.Character:SetPrimaryPartCFrame(chest.CFrame)
			wait(0.3) -- Ждем 0.3 секунды перед перемещением к следующему сундуку
		else
			print("Объект " .. chestName .. " не найден в Workspace!")
		end
	end
end

-- Обработчик нажатия на кнопку
button1.MouseButton1Click:Connect(function()
	MoveToChests()
end)


button2.MouseButton1Click:Connect(function()
	-- Получаем объекты "PirateSloop", "Dinghy" и "PirateBrigade" из папки Workspace.Boats
	local boatsFolder = game.Workspace.Boats
	if boatsFolder then
		local pirateSloop = boatsFolder:FindFirstChild("PirateSloop")
		local dinghy = boatsFolder:FindFirstChild("Dinghy")
		local pirateBrigade = boatsFolder:FindFirstChild("PirateBrigade")

		-- Перемещаем игрока к первому доступному кораблю
		if pirateSloop then
			game.Players.LocalPlayer.Character:SetPrimaryPartCFrame(pirateSloop.CFrame)
		elseif dinghy then
			game.Players.LocalPlayer.Character:SetPrimaryPartCFrame(dinghy.CFrame)
		elseif pirateBrigade then
			game.Players.LocalPlayer.Character:SetPrimaryPartCFrame(pirateBrigade.CFrame)
		else
			print("Не удалось найти корабли в Workspace.Boats")
		end
	else
		print("Папка Boats не найдена в Workspace!")
	end
end)

-- Обработчик нажатия на кнопку
button3.MouseButton1Click:Connect(function()
	-- Получаем все объекты с именем "Fruit" в папке Workspace
	local fruits = game.Workspace:GetChildren()
	for _, fruit in ipairs(fruits) do
		if fruit.Name == "Fruit" then
			-- Перемещаем игрока к объекту с именем "Fruit"
			game.Players.LocalPlayer.Character:SetPrimaryPartCFrame(fruit.CFrame)
			wait(1) -- Ждем 1 секунду перед перемещением к следующему объекту "Fruit"
		end
	end
end)
