local function CreateButton(text, position, parent)

end

local function MoveToCollectibles()
	local collectiblesFolder = game.Workspace:WaitForChild("collectibles") -- Находим папку "collectibles" в Workspace
	local collectibleParts = collectiblesFolder:GetChildren() -- Получаем все дочерние объекты в папке "collectibles"

	for _, collectiblePart in ipairs(collectibleParts) do -- Перебираем все части в папке
		if collectiblePart.Name == "С" and collectiblePart:IsA("BasePart") then -- Проверяем, что имя объекта равно "Partc" и он является частью
			game.Players.LocalPlayer.Character:MoveTo(collectiblePart.Position) -- Перемещаем игрока к позиции объекта
			wait(2) -- Ждем 2 секунды перед перемещением к следующему объекту
		end
	end
end

buttonFarm1.MouseButton1Click:Connect(function()
	MoveToCollectibles() -- Вызываем функцию при нажатии на кнопку
end)

local function TeleportToRoyalJelly()
	-- Получаем объект RoyalJelly
	local royalJelly = game.Workspace.Shops.JellyDispenser.Items.RoyalJelly
	if royalJelly then
		-- Перемещаем игрока к RoyalJelly
		player.Character:MoveTo(royalJelly.Position)
	else
		warn("Объект RoyalJelly не найден.")
	end
end

local function MoveToCparts()
	local collectiblesFolder = game.Workspace:FindFirstChild("collectibles")
	if collectiblesFolder then
		local c = collectiblesFolder:GetChildren()
		for _, c in ipairs(c) do
			if c.Name == "C" then
				game.Players.LocalPlayer.Character:MoveTo(c.Position)
				wait(0.2)
			end
		end
	else
		warn("Collectibles folder not found in Workspace.")
	end
end
